# –í–æ—Å–ø–æ–ª—å–∑—É–µ–º—Å—è asincio

```python
import asyncio
import time
async def fun1(x):
    print(x**2)
    await asyncio.sleep(2)
    print('fun1 –∑–∞–≤–µ—Ä—à–µ–Ω–∞')
async def fun2(x):
    print(x**3)
    await asyncio.sleep(2)
    print('fun2 –∑–∞–≤–µ—Ä—à–µ–Ω–∞')

async def main():
    task1 = asyncio.create_task(fun1(4))
    task2 = asyncio.create_task(fun2(4))

    await task1
    await task2

print(time.strftime('%X'))
asyncio.run(main())
print(time.strftime('%X'))
```
